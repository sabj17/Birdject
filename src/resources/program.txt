class Output {
  public:
    int pin;
    int state = 0;

    Output(int inputPin) {
      pin = inputPin;
    }
    void setupClass() {
      pinMode(pin, OUTPUT);
    }


    void setState(int newState) {
      state = newState;
      digitalWrite(pin, newState);
    }

    int getState(){
      return state;
    }

    void switchState() {
      if (state) {
        state = 0;
      } else {
        state = 1;
      }
      digitalWrite(pin, state);
      delay(200);
    }
};

class Light : public Output {
  public:
    Light (int inputPin): Output(inputPin) {}
};

class Input {
  public:
    int pin;

    Input (int inputPin) {
      pin = inputPin;
    }

    void setupClass() {
      pinMode(pin, INPUT);
    }

    int Read() {
      return digitalRead(pin);
    }
};


class Switch : public Input {
  public:
    Switch (int inputPin): Input(inputPin) {}
};



class LivingRoomClass {
  public:
	Light light1;
	Switch switch1;
	int y = 10;
	LivingRoomClass() : light1(pin7) , switch1(pin2) {}

} LivingRoom;



class KitchenClass {
  public:
	Light light2;
	Switch switch2;
	KitchenClass() : light2(pin8) , switch2(pin3) {}

} Kitchen;

bool x = false;

void setup() {
	Serial.begin(9600);
	LivingRoom.light1.setupClass();
	LivingRoom.switch1.setupClass();
	Kitchen.light2.setupClass();
	Kitchen.switch2.setupClass();
}

void loop() {

	if (LivingRoom.switch1.Read()){
	LivingRoom.light1.switchState();
	x = 	LivingRoom.light1.getState();

	if (x == true) {
		Serial.println(String("LivingRoom Light is on"));
	}
	else {
		Serial.println(String("LivingRoom Light is off"));
	}
	x = false;
	}

	if (Kitchen.switch2.Read()){
	Kitchen.light2.switchState();
	x = 	Kitchen.light2.getState();

	if (x == true) {
		Serial.println(String("Kitchen Light is on"));
	}
	else {
		Serial.println(String("Kitchen Light is off"));
	}
	x = false;
	}
}
