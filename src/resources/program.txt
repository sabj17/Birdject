class Output {
  public:
    int pin;
    int state = 0;

    Output(int inputPin) {
      pin = inputPin;
    }
    void initialize() {
      pinMode(pin, OUTPUT);
    }


    void setMode(bool newMode) {
      state = newMode;
      digitalWrite(pin, newMode);
    }

    int isTurnedOn(){
      return state;
    }

    int changeMode() {
      if (state) {
        state = 0;
      } else {
        state = 1;
      }
      digitalWrite(pin, state);
      delay(200);
      return state;
    }
};

class Light : public Output {
  public:
    Light (int inputPin): Output(inputPin) {}
};

class Window : public Output {
  public:
    Window (int inputPin): Output(inputPin) {}
};

class Radiator : public Output {
  public:
    Radiator (int inputPin): Output(inputPin) {}

    int increase(){
      if(state < 5){
        state++;
      }
      return state;
    }

    int decrease(){
      if(state > 0){
        state--;
      }
      return state;
    }

};



class Input {
  public:
    int pin;

    Input (int inputPin) {
      pin = inputPin;
    }

    void initialize() {
      pinMode(pin, INPUT);
    }

    int isTurnedOn() {
      return digitalRead(pin);
    }
};


class Switch : public Input {
  public:
    Switch (int inputPin): Input(inputPin) {}
};

class Thermometer : public Input {
  public:
    Thermometer (int inputPin): Input(inputPin) {}

    float isTurnedOn() {
      return analogRead(pin);
    }

    float getTemp(){
        DHT.read11(pin);
        return DHT.temperature;
    }
};

class LivingRoomClass {
  public:
	Light light1;
	Light light2;

	void switch_all_lights (){
	light1.changeMode();
	light2.changeMode();
	}

	LivingRoomClass() : light1(pin8) , light2(pin9) {}

} LivingRoom;

Switch switch1(pin2);

void initializeObjects(){
	LivingRoom.light1.initialize();
	LivingRoom.light2.initialize();
	switch1.initialize();
}

void setup() {
	Serial.begin(9600);
	initializeObjects();
}

void loop() {

	if (switch1.isTurnedOn()){
	LivingRoom.switch_all_lights();
	}
}
