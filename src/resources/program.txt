class Output {
  public:
    int pin;
    int state = 0;

    Output(int inputPin) {
      pin = inputPin;
    }
    void initialize() {
      pinMode(pin, OUTPUT);
    }


    void setMode(int newState) {
      state = newState;
      digitalWrite(pin, newState);
    }

    int isTurnedOn(){
      return state;
    }

    int changeMode() {
      if (state) {
        state = 0;
      } else {
        state = 1;
      }
      digitalWrite(pin, state);
      delay(200);
      return state;
    }
};

class Light : public Output {
  public:
    Light (int inputPin): Output(inputPin) {}
};

class Window : public Output {
  public:
    Window (int inputPin): Output(inputPin) {}
};

class Radiator : public Output {
  public:
    Radiator (int inputPin): Output(inputPin) {}
};



class Input {
  public:
    int pin;

    Input (int inputPin) {
      pin = inputPin;
    }

    void initialize() {
      pinMode(pin, INPUT);
    }

    int isTurnedOn() {
      return digitalRead(pin);
    }
};


class Switch : public Input {
  public:
    Switch (int inputPin): Input(inputPin) {}
};

class Thermometer : public Input {
  public:
    Thermometer (int inputPin): Input(inputPin) {}

    float isTurnedOn() {
        DHT.read11(pin);
        return DHT.temperature;
    }

    float getTemp(){
        DHT.read11(pin);
        return DHT.temperature;
    }
};
